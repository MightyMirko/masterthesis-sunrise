148  /**
149  * \brief FRI client interface.
150  *
151  * This is the callback interface that should be implemented by FRI clients.
152  * Callbacks are automatically called by the client application
153  * (ClientApplication) whenever new FRI messages arrive.
154  */
155  class IClient
156  {
157  friend class ClientApplication;
158 
159  public:
160 
161  /** \brief Virtual destructor. */
162  virtual ~IClient() {}
163 
164  /**
165  * \brief Callback that is called whenever the FRI session state changes.
166  *
167  * @param oldState previous FRI session state
168  * @param newState current FRI session state
169  */
170  virtual void onStateChange(ESessionState oldState, ESessionState newState) = 0;
171 
172  /**
173  * \brief Callback for the FRI session states 'Monitoring Wait' and 'Monitoring Ready'.
174  */
175  virtual void monitor() = 0;
176 
177  /**
178  * \brief Callback for the FRI session state 'Commanding Wait'.
179  */
180  virtual void waitForCommand() = 0;
181 
182  /**
183  * \brief Callback for the FRI session state 'Commanding'.
184  */
185  virtual void command() = 0;
186 
187  protected:
188 
189  /**
190  * \brief Method to create and initialize the client data structure (used internally).
191  *
192  * @return newly allocated client data structure
193  */
194  virtual ClientData* createData() = 0;
