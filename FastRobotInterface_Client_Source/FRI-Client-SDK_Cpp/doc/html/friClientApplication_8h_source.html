77  /**
78  * \brief FRI client application class.
79  *
80  * A client application takes an instance of the IConnection interface and
81  * an instance of an IClient interface to provide the functionality
82  * needed to set up an FRI client application. It can be used to easily
83  * integrate the FRI client code within other applications.
84  * The algorithmic functionality of an FRI client application is implemented
85  * using the IClient interface.
86  */
87  class ClientApplication
88  {
89 
90  public:
91 
92  /**
93  * \brief Constructor without transformation client.
94  *
95  * This constructor takes an instance of the IConnection interface and
96  * an instance of the IClient interface as parameters.
97  * @param connection FRI connection class
98  * @param client FRI client class
99  */
100  ClientApplication(IConnection& connection, IClient& client);
101 
102  /**
103  * \brief Constructor with transformation client.

105  * This constructor takes an instance of the IConnection interface and
106  * an instance of the IClient interface and an instance of a
107  * TransformationClient as parameters.
108  * @param connection FRI connection class
109  * @param client FRI client class
110  * @param trafoClient FRI transformation client class
111  */
112  ClientApplication(IConnection& connection, IClient& client, TransformationClient& trafoClient);
113 
114  /** \brief Destructor. */

117  /**

122  * @return True if connection was established
123  */
124  bool connect(int port, const char *remoteHost = NULL);
