82  /**
83  * \brief Set the joint positions for the current interpolation step.
84  *
85  * This method is only effective when the client is in a commanding state.
86  * @param values Array with the new joint positions (in rad)
87  */
88  void setJointPosition(const double* values);
89 
90  /**
91  * \brief Set the applied wrench vector of the current interpolation step.
92  *
93  * The wrench vector consists of:
94  * [F_x, F_y, F_z, tau_A, tau_B, tau_C]
95  *
96  * F ... forces (in N) applied along the Cartesian axes of the
97  * currently used motion center.
98  * tau ... torques (in Nm) applied along the orientation angles
99  * (Euler angles A, B, C) of the currently used motion center.
100  *
101  * This method is only effective when the client is in a commanding state.
102  * The ControlMode of the robot has to be Cartesian impedance control mode. The
103  * Client Command Mode has to be wrench.
104  *
105  * @param wrench Applied Cartesian wrench vector.
106  */
107  void setWrench(const double* wrench);
108 
109  /**
110  * \brief Set the applied joint torques for the current interpolation step.
111  *
112  * This method is only effective when the client is in a commanding state.
113  * The ControlMode of the robot has to be joint impedance control mode. The
114  * Client Command Mode has to be torque.
115  *
116  * @param torques Array with the applied torque values (in Nm)
117  */
118  void setTorque(const double* torques);
119 
120  /**
121  * \brief Set boolean output value.
122  *
123  * @throw FRIException Throws a FRIException if more outputs are set than can be registered.
124  * @throw FRIException May throw an FRIException if the IO is of wrong type, unknown or not an output.
125  * @param name Full name of the IO (Syntax "IOGroupName.IOName").
126  * @param value Boolean value to set.
127  */
128  void setBooleanIOValue(const char* name, const bool value);
129 
130  /**
131  * \brief Set digital output value.
132  *
133  * @throw FRIException Throws a FRIException if more outputs are set than can be registered.
134  * @throw FRIException May throw an FRIException if the IO is of wrong type, unknown or not an output.
135  * @param name Full name of the IO (Syntax "IOGroupName.IOName").
136  * @param value Digital value to set.
137  */
138  void setDigitalIOValue(const char* name, const unsigned long long value);
139 
140  /**
141  * \brief Set analog output value.
142  *
143  * @throw FRIException Throws a FRIException if more outputs are set than can be registered.
144  * @throw FRIException May throw an FRIException if the IO is of wrong type, unknown or not an output.
145  * @param name Full name of the IO (Syntax "IOGroupName.IOName").
146  * @param value Analog value to set.
147  */
148  void setAnalogIOValue(const char* name, const double value);
149
